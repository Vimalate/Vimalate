## 场景一：根据 status 显示对应名称

### 使用对象
```js
const statusStr={
  '1':'待付款',
  '2': '待发货',
  '3': '已发货',
  '4': '交易完成',
  '5': '交易关闭',
  'default': '',
}

const getStatus=(status)=>{
  return statusStr[status]||statusStr['default']
}
```
### 利用 map
```js
const statusStr = new Map([
  ['1','待付款'],
  ['2','待发货'],
  ['3','已发货'],
  ['4','交易完成'],
  ['5','交易关闭'],
  ['default',''],
])
const getStatus = (status) =>{
  let actions = statusStr.get(status) || statusStr.get('default')
  return  actions;
}
```
